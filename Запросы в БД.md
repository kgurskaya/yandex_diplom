#### Задача 1
```
SELECT
    c.login,
    COUNT (*)
FROM
    "Couriers" AS c
    JOIN "Orders" AS o on (o."courierId"=c.id)
WHERE
    o."inDelivery" = true
GROUP BY
    c.login;
```
#### Задача 2
```
SELECT
    track,
    CASE
        WHEN finished = true THEN 2
        WHEN cancelled = true THEN -1
        WHEN "inDelivery" = true THEN 1
        ELSE 0
    END AS status
FROM
    "Orders";
```
